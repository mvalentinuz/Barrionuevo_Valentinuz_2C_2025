188 ble_config_t BLEconfig = {
189 "AppTender",
190 BLE_NO_INT
191 };
192 // Inicializaciones
193 AnalogInputInit(&analogInputConfig);
194 TimerInit(&timer_config);
195 ServoInit(SERVO_TENDER, SERVO_PIN);
196 LedsInit();
197 SwitchesInit();

200
201 // Creación de tareas

204 xTaskCreate(&bluetooth, "Bluetooth", 4096, NULL, 5, &bluetooth_task_handle);
205
206 // Inicio del timer
207 TimerStart(timer_config.timer);
